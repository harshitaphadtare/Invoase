import express from 'express';
import multer from 'multer';
import {
    uploadBill,
    getBillsByDocument,
    deleteBill,
    updateBillStatus
} from '../controllers/billController.js';
import { authStudent } from '../middleware/AuthStudent.js';

const router = express.Router();

// Configure multer for memory storage
const storage = multer.memoryStorage();
const upload = multer({
    storage,
    limits: {
        fileSize: 5 * 1024 * 1024, // 5MB limit
    },
    fileFilter: (req, file, cb) => {
        const allowedTypes = ['image/jpeg', 'image/png', 'application/pdf'];
        if (allowedTypes.includes(file.mimetype)) {
            cb(null, true);
        } else {
            cb(new Error('Invalid file type. Only JPG, PNG and PDF are allowed'));
        }
    }
});

// Routes
router.post('/upload', authStudent, upload.single('bill'), uploadBill);
router.get('/document/:documentId', authStudent, getBillsByDocument);
router.delete('/:billId', authStudent, deleteBill);
router.patch('/:billId/status', authStudent, updateBillStatus);

export default router;
